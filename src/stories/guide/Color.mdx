# Usage

### 1. Surface

<table id="guide" style={{ width: "100%", tableLayout: "fixed", margin: "40px 0" }}>
  <thead>
    <tr>
      <th style={{ width: "100px" }}>-</th>
      <th>Color Style Name</th>
      <th>Color Palette</th>
      <th>Hex</th>
      <th style={{ width: "42%" }}>Description</th>
    </tr>
  </thead>
  <tbody>
    {Object.keys(color?.surface).map((surfaceStyleNm, index) => {
      const surfaceColorObj = color?.surface[surfaceStyleNm] || {};
      return (
        <tr key={index}>
          <td style={{ backgroundColor: surfaceColorObj.hex, height: "100px" }} />
          <td className="bold">{surfaceColorObj.styleNm}</td>
          <td>{`${surfaceColorObj.palette}/${surfaceColorObj.depth}`}</td>
          <td>{surfaceColorObj.hex}</td>
          <td style={{ textAlign: "left" }}>
            {surfaceColorObj.description?.split("\n")?.map((line) => <p>{line}</p> || <br />)}
          </td>
        </tr>
      );
    })}
  </tbody>
</table>
